API Gateways

GraphQL Approach

GraphQL

  • A data query language
  • Developed by Facebook
  • Used internally since 2012
  • Open source version published in July 2015
  • Specification: https://facebook.github.io/graphql

Response Structure


query {
  title(titleNumber:"AGL10368"){
    titleNumber,
    classOfTitle,
    estateInterest
  }
}
          

{

Type System


type Title {
  applicationReference: String!
  applicationTimestamp: OffsetDateTime
  classOfTitle: ClassOfTitle
  editionDate: LocalDate
  entries: [RegisterEntry]!
  estateInterest: EstateInterest
  status: TitleStatusInfo!
  tenure: Tenure
  titleNumber: String!
}

Mutations & Subscriptions


mutation ChangeTitle {
  changeStatus(titleNumber: AGL10368, newStatus: "Closed", date: "2018-01-02" ) {
    titleNumber, status
  }

  setApplicationReference(titleNumber: "AGL10368", newApplicationReference: "ABR33434") {
    titleNumber, status, applicationReference
  }
}

subscription Entries {
  entryChanges(titleNumber: "AGL10368") { entries }
  entryDeleted { entryId }
}

Example: JavaScript


var ProductType = new GraphQLObjectType({
  name: "Product",
  description: "Commodity available for purchase",
  fields: {
    id: {type: GraphQLString},
    onStock: {
      type: GraphQLInt,
      resolve(product) {
        return product.onStock + 10;
      }
    }
  }
});
